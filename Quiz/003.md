in_protein=str(input("Enter Protein Input:"))
protein1=""
for z in range(0,len(in_protein)):
    if in_protein[z]=="A":
        protein1=protein1+"T"
    if in_protein[z]=="G":
        protein1=protein1+"C"
    if in_protein[z]=="T":
        protein1=protein1+"A"
    if in_protein[z]=="C":
        protein1=protein1+"G"
print(protein1)
